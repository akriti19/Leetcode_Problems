class Solution {
    public List<Boolean> kidsWithCandies(int[] candies, int extraCandies) {
           ArrayList <Boolean> result = new ArrayList<>();
           int max = candies[0];
           for(int i=0 ; i<candies.length;i++)
           {
               if(candies[i]>=max)
               {
                   max=candies[i];
               }
           }

           for(int j=0 ; j<candies.length;j++)
           {
               int sum = candies[j]+extraCandies;
               if(sum>=max)
               {
                   result.add(true);
               }
               else
               {
                   result.add(false);
               }
           }

           return result;
    }
}
